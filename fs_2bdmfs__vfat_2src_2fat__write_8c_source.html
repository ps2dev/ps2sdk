33  reorder (swap) the cluster stack records
34 */
35 static void swapClStack(fat_driver* fatd, int startIndex, int endIndex)
36 {
37  int i;
38  int size;
39  int offset1, offset2;
40  unsigned int tmp;
41 
42  size = endIndex - startIndex;
43  if (size < 2) {
44  return;
45  }
46 
47  size /= 2;
48  for (i = 0; i < size; i++) {
49  offset1 = startIndex + i;
50  offset2 = endIndex - 1 - i;
51  tmp = fatd->clStack[offset1];
52  fatd->clStack[offset1] = fatd->clStack[offset2];
53  fatd->clStack[offset2] = tmp;
54  }
55 }

62 static int fat_readEmptyClusters12(fat_driver* fatd)
63 {
64  int ret;
65  int recordOffset;
66  int sectorSpan;
67  int fatSector;
68  int lastFatSector;
69  unsigned int cluster;
70  unsigned int clusterValue;
71  unsigned char xbuf[4];
72  int oldClStackIndex;
73  unsigned char* sbuf = NULL; //sector buffer
74 
75  oldClStackIndex = fatd->clStackIndex;
76 
77  lastFatSector = -1;
78  cluster = fatd->clStackLast;
79 
80  while (fatd->clStackIndex < MAX_CLUSTER_STACK) {
81  recordOffset = (cluster * 3) / 2; //offset of the cluster record (in bytes) from the FAT start
82  fatSector = recordOffset / fatd->partBpb.sectorSize;
83  sectorSpan = 0;
84  if ((recordOffset % fatd->partBpb.sectorSize) == (fatd->partBpb.sectorSize - 1)) {
85  sectorSpan = 1;
86  }
87  if (lastFatSector != fatSector || sectorSpan) {
88  ret = READ_SECTOR(fatd, fatd->partBpb.partStart + fatd->partBpb.resSectors + fatSector, sbuf);
89  if (ret < 0) {
90  M_DEBUG("Read fat12 sector failed! sector=%u! \n", fatd->partBpb.partStart + fatd->partBpb.resSectors + fatSector);
91  return -EIO;
92  }
93  lastFatSector = fatSector;
94 
95  if (sectorSpan) {
96  xbuf[0] = sbuf[fatd->partBpb.sectorSize - 2];
97  xbuf[1] = sbuf[fatd->partBpb.sectorSize - 1];
98  ret = READ_SECTOR(fatd, fatd->partBpb.partStart + fatd->partBpb.resSectors + fatSector + 1, sbuf);
99  if (ret < 0) {
100  M_DEBUG("Read fat12 sector failed sector=%u! \n", fatd->partBpb.partStart + fatd->partBpb.resSectors + fatSector + 1);
101  return -EIO;
102  }
103  xbuf[2] = sbuf[0];
104  xbuf[3] = sbuf[1];
105  }
106  }
107  if (sectorSpan) { // use xbuf as source buffer
108  clusterValue = fat_getClusterRecord12(xbuf + (recordOffset % fatd->partBpb.sectorSize) - (fatd->partBpb.sectorSize - 2), cluster % 2);
109  } else { // use sector buffer as source buffer
110  clusterValue = fat_getClusterRecord12(sbuf + (recordOffset % fatd->partBpb.sectorSize), cluster % 2);
111  }
112  if (clusterValue == 0) {
113  fatd->clStackLast = cluster;
114  fatd->clStack[fatd->clStackIndex] = cluster;
115  fatd->clStackIndex++;
116  }
117  cluster++; //read next cluster record in the sequence
118  }
119  //the stack operates as LIFO but we put in the clusters as FIFO
120  //we should reverse the cluster order - not necessary
121  //but it will retain the natural (increasing) order of
122  //the cluster chain
123  swapClStack(fatd, oldClStackIndex, fatd->clStackIndex);
124  return fatd->clStackIndex;
125 }

300 /*
301  save value at the cluster record in FAT 12
302 */
303 static int fat_saveClusterRecord12(fat_driver* fatd, unsigned int currentCluster, unsigned int value)
304 {
305  int ret;
306  int sectorSpan;
307  int recordOffset;
308  int recordType;
309  int fatNumber;
310  unsigned int fatSector;
311 
312  ret = -1;
313  //recordOffset is byte offset of the record from the start of the fat table
314  recordOffset = (currentCluster * 3) / 2;
315 
316  //save both fat tables
317  for (fatNumber = 0; fatNumber < fatd->partBpb.fatCount; fatNumber++) {
318  unsigned char* sbuf = NULL; //sector buffer
319 
320  fatSector = fatd->partBpb.partStart + fatd->partBpb.resSectors + (fatNumber * fatd->partBpb.fatSize);
321  fatSector += recordOffset / fatd->partBpb.sectorSize;
322  sectorSpan = fatd->partBpb.sectorSize - (recordOffset % fatd->partBpb.sectorSize);
323  if (sectorSpan > 1) {
324  sectorSpan = 0;
325  }
326  recordType = currentCluster % 2;
327 
328  ret = READ_SECTOR(fatd, fatSector, sbuf);
329  if (ret < 0) {
330  M_DEBUG("Read fat16 sector failed! sector=%u! \n", fatSector);
331  return -EIO;
332  }
333  if (!sectorSpan) { // not sector span - the record is copmact and fits in single sector
334  fat_setClusterRecord12(sbuf + (recordOffset % fatd->partBpb.sectorSize), value, recordType);
335  ret = WRITE_SECTOR(fatd, fatSector);
336  if (ret < 0) {
337  M_DEBUG("Write fat12 sector failed! sector=%u! \n", fatSector);
338  return -EIO;
339  }
340  } else { // sector span - the record is broken in 2 pieces - each one on different sector
341  //modify one last byte of the sector buffer
342  fat_setClusterRecord12part1(sbuf + (recordOffset % fatd->partBpb.sectorSize), value, recordType);
343  //save current sector
344  ret = WRITE_SECTOR(fatd, fatSector);
345  if (ret < 0) {
346  M_DEBUG("Write fat12 sector failed! sector=%u! \n", fatSector);
347  return -EIO;
348  }
349  //read next sector from the fat
350  fatSector++;
351  ret = READ_SECTOR(fatd, fatSector, sbuf);
352  if (ret < 0) {
353  M_DEBUG("Read fat16 sector failed! sector=%u! \n", fatSector);
354  return -EIO;
355  }
356  //modify first byte of the sector buffer
357  fat_setClusterRecord12part2(sbuf, value, recordType);
358  //save current sector
359  ret = WRITE_SECTOR(fatd, fatSector);
360  if (ret < 0) {
361  M_DEBUG("Write fat12 sector failed! sector=%u! \n", fatSector);
362  return -EIO;
363  }
364  }
365  } //end for
366  return ret;
367 }

454 //---------------------------------------------------------------------------
455 /*
456  Append (and write) cluster chain to the FAT table.
457 
458  currentCluster - current end cluster record.
459  endCluster - new end cluster record.
460 
461  Note: there is no checking wether the currentCluster holds the EOF marker!
462 
463  Example
464  current FAT:
465 
466  index 09 10 11 12
467  +-----+-----+-----+-----+
468  value + 10 | EOF | 0 | 0 |
469  +-----+-----+-----+-----+
470 
471 
472  currentcluster = 10, endcluster = 12
473  updated FAT (after the function ends):
474 
475  index 09 10 11 12
476  +-----+-----+-----+-----+
477  value + 10 | 12 | 0 | EOF |
478  +-----+-----+-----+-----+
479 
480 */
481 //---------------------------------------------------------------------------
482 static int fat_appendClusterChain(fat_driver* fatd, unsigned int currentCluster, unsigned int endCluster)
483 {
484  int ret;
485  ret = -1;
486  switch (fatd->partBpb.fatType) {
487  case FAT12:
488  ret = fat_saveClusterRecord12(fatd, currentCluster, endCluster);
489  if (ret < 0)
490  return ret;
491  ret = fat_saveClusterRecord12(fatd, endCluster, 0xFFF);
492  break;
493 
494  case FAT16:
495  ret = fat_saveClusterRecord16(fatd, currentCluster, endCluster);
496  if (ret < 0)
497  return ret;
498  ret = fat_saveClusterRecord16(fatd, endCluster, 0xFFFF);
499  break;
500 
501  case FAT32:
502  ret = fat_saveClusterRecord32(fatd, currentCluster, endCluster);
503  if (ret < 0)
504  return ret;
505  ret = fat_saveClusterRecord32(fatd, endCluster, 0xFFFFFFF);
506  break;
507  }
508  return ret;
509 }

548 */
549 static int fat_deleteClusterChain(fat_driver* fatd, unsigned int cluster)
550 {
551  int ret;
552  int size;
553  int cont;
554  int end;
555  int i;
556 
557  if (cluster < 2) {
558  return -EFAULT;
559  }
560  M_DEBUG("I: delete cluster chain starting at cluster=%u\n", cluster);
561 
562  cont = 1;
563 
564  while (cont) {
565  size = fat_getClusterChain(fatd, cluster, fatd->cbuf, MAX_DIR_CLUSTER, 1);
566 
567  end = size - 1; //do not delete last cluster in the chain buffer
568

570  ret = fat_modifyClusterChain(fatd, fatd->cbuf[i], 0);

572  return ret;

577  cluster = fatd->cbuf[end];
578  } else {
579  //no more cluster entries - delete the last cluster entry
580  ret = fat_modifyClusterChain(fatd, fatd->cbuf[end], 0);
581  if (ret < 0) {
582  return ret;
583  }
584  cont = 0;
585  }
586  fat_invalidateLastChainResult(fatd); //prevent to misuse current (now deleted) fatd->cbuf
587  }
588  return 1;
589 }

598 {
599

602
