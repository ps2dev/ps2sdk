43int initHubDriver(void)
44{
45 int i;
46 UsbHub *hub;
47 u32 needMem = usbConfig.maxHubDevices * sizeof(UsbHub);
48
49 hubBufferList = (UsbHub *)AllocSysMemory(ALLOC_FIRST, needMem, 0);
50 if (!hubBufferList) {
51 dbg_printf("ERROR: unable to alloc hub buffer\n");
52 return -1;
53 }
54 memset(hubBufferList, 0, needMem);
55
56 hub = hubBufferList;
57 for (i = 0; i < usbConfig.maxHubDevices - 1; i++) {
58 hub->next = hub + 1;
59 hub++;
60 }
61 hub->next = NULL;
62
63#if USE_GP_REGISTER
64 doRegisterDriver(&HubDriver, &_gp);
65#else
66 doRegisterDriver(&HubDriver, NULL);
67#endif
68 return 0;
69}

107void hubStatusChangeCallback(IoRequest *req)
108{
109 UsbHub *dev = (UsbHub *)req->userCallbackArg;
110
111 if (req->resultCode == USB_RC_OK) {
112 if (dev->statusChangeInfo[0] & 1) {
113 dev->statusChangeInfo[0] &= ~1;
114 HubControlTransfer(dev,
115 USB_DIR_IN | USB_RT_HUB, USB_REQ_GET_STATUS, 0, 0, 4, &dev->hubStatus,
116 hubGetHubStatusCallback);
117 } else {
118 int port;
119
120 if (dev->hubStatusCounter > 0) {
121 port = dev->hubStatusCounter;
122 if (dev->statusChangeInfo[port >> 3] & BIT(port & 7)) {
123 dev->statusChangeInfo[port >> 3] &= ~BIT(port & 7);
124 dev->portCounter = port;
125 HubControlTransfer(dev,
126 USB_DIR_IN | USB_RT_PORT, USB_REQ_GET_STATUS, 0, port, 4, &dev->portStatusChange,
127 hubGetPortStatusCallback);
128 return;
129 }
130 } else {
131 for (port = 1; (u32)port <= dev->numChildDevices; port++) {
132 if (dev->statusChangeInfo[port >> 3] & BIT(port & 7)) {
133 dev->statusChangeInfo[port >> 3] &= ~BIT(port & 7);
134 dev->portCounter = port;
135 HubControlTransfer(dev,
136 USB_DIR_IN | USB_RT_PORT, USB_REQ_GET_STATUS, 0, port, 4, &dev->portStatusChange,
137 hubGetPortStatusCallback);
138 return;
139 }
140 }
141 }
142 getHubStatusChange(dev);
143 }
144 } else
145 dbg_printf("hubStatusChangeCallback, iores %d\n", req->resultCode);
146}

169int cancelTimerCallback(TimerCbStruct *arg)
170{
171 if (arg->isActive) {
172 if (arg->next)
173 arg->next->prev = arg->prev;
174 else
175 memPool.timerListEnd = arg->prev;
176
177 if (arg->prev)
178 arg->prev->next = arg->next;
179 else
180 memPool.timerListStart = arg->next;
181
182 arg->prev = arg->next = NULL;
183 arg->isActive = 0;
184 return 0;
185 } else
186 return -1;
187}

189int addTimerCallback(TimerCbStruct *arg, TimerCallback func, void *cbArg, u32 delay)
190{
191 if (arg->isActive)
192 return -1;
193
194 arg->isActive = 1;
195 arg->callbackProc = func;
196 arg->callbackArg = cbArg;
197
198 TimerCbStruct *pos = memPool.timerListStart;
199
200 while (pos && (delay > pos->delayCount)) {
201 delay -= pos->delayCount;
202 pos = pos->next;
203 }
204 if (pos) {
205 arg->prev = pos->prev;
206 if (pos->prev)
207 pos->prev->next = arg;
208 else
209 memPool.timerListStart = arg;
210
211 arg->next = pos;
212 pos->prev = arg;
213 pos->delayCount -= delay;
214 } else {
215 arg->prev = memPool.timerListEnd;
216 if (memPool.timerListEnd)
217 memPool.timerListEnd->next = arg;
218 else
219 memPool.timerListStart = arg;
220 memPool.timerListEnd = arg;
221 arg->next = NULL;
222 }
223 arg->delayCount = delay;
224 memPool.ohciRegs->HcInterruptEnable = OHCI_INT_SF;
225 return 0;
226}

228void killEndpoint(Endpoint *ep)
229{
230 int i = 0;
231 IoRequest *req;
232 HcED *hcEd = &ep->hcEd;
233 if (ep->endpointType == TYPE_ISOCHRON) {
234 for (i = 0; i < usbConfig.maxIsoTransfDesc; i++) {
235 req = memPool.hcIsoTdToIoReqLUT[i];
236 if (req && (req->correspEndpoint == ep)) {
237 freeIoRequest(req);
238 memPool.hcIsoTdToIoReqLUT[i] = NULL;
239 freeIsoTd(memPool.hcIsoTdBuf + i);
240 }
241 }
242 freeIsoTd((HcIsoTD *)hcEd->tdTail);
243 } else {
244 for (i = 0; i < usbConfig.maxTransfDesc; i++) {
245 req = memPool.hcTdToIoReqLUT[i];
246 if (req && (req->correspEndpoint == ep)) {
247 freeIoRequest(req);
248 memPool.hcTdToIoReqLUT[i] = NULL;
249 freeTd(memPool.hcTdBuf + i);
250 }
251 }
252 freeTd(hcEd->tdTail);
253 }
254 hcEd->tdTail = NULL;
255 hcEd->tdHead = NULL;
256
257 while ((req = ep->ioReqListStart)) {
258 if (req->next)
259 req->next->prev = req->prev;
260 else
261 ep->ioReqListEnd = req->prev;
262
263 if (req->prev)
264 req->prev->next = req->next;
265 else
266 ep->ioReqListStart = req->next;
267
268 freeIoRequest(req);
269 }
270 removeEndpointFromQueue(ep);
271
272 ep->next = NULL;
273 ep->prev = memPool.freeEpListEnd;
274 if (memPool.freeEpListEnd)
275 memPool.freeEpListEnd->next = ep;
276 else
277 memPool.freeEpListStart = ep;
278 memPool.freeEpListEnd = ep;
279}

281int removeEndpointFromDevice(Device *dev, Endpoint *ep)
282{
283 ep->hcEd.hcArea |= HCED_SKIP;
284 removeHcEdFromList(ep->endpointType, &ep->hcEd);
285
286 if (ep->next)
287 ep->next->prev = ep->prev;
288 else
289 dev->endpointListEnd = ep->prev;
290
291 if (ep->prev)
292 ep->prev->next = ep->next;
293 else
294 dev->endpointListStart = ep->next;
295
296 ep->correspDevice = NULL;
297 addTimerCallback(&ep->timer, (TimerCallback)killEndpoint, ep, 200);
298 return 0;
299}

309void hubResetDevice(void *devp)
310{
311 Device *dev = devp;
312 if (memPool.delayResets) {
313 dev->deviceStatus = DEVICE_RESETDELAYED;
314 } else {
315 memPool.delayResets = 1;
316 dev->deviceStatus = DEVICE_RESETPENDING;
317 dev->resetFlag = 1;
318 if (dev->parent == memPool.deviceTreeRoot) { // root hub port
319 memPool.ohciRegs->HcRhPortStatus[dev->attachedToPortNo - 1] = BIT(PORT_RESET);
320 } else { // normal hub port
321 UsbHub *hub;
322
323 hub = (UsbHub *)dev->parent->privDataField;
324 hub->hubStatusCounter = dev->attachedToPortNo;
325
326 HubControlTransfer(hub,
327 USB_DIR_OUT | USB_RT_PORT, USB_REQ_SET_FEATURE, PORT_RESET, dev->attachedToPortNo, 0, NULL,
328 hubDeviceResetCallback);
329 }
330 }

333int checkDelayedResetsTree(Device *tree)
334{
335 Device *dev;
336 for (dev = tree->childListStart; dev != NULL; dev = dev->next) {
337 if (dev->deviceStatus == DEVICE_RESETDELAYED) {
338 hubResetDevice(dev);
339 return 1;
340 }
341 if ((dev->deviceStatus == DEVICE_READY) && dev->childListStart) {
342 if (checkDelayedResetsTree(dev))
343 return 1;
344 }
345 }
346 return 0;
347}

364void flushPort(Device *dev)
365{
366 Device *child;
367 if (dev->deviceStatus != DEVICE_NOTCONNECTED) {
368 dev->deviceStatus = DEVICE_NOTCONNECTED;
369 if (dev->devDriver) {
370 callUsbDriverFunc(dev->devDriver->disconnect, dev->id, dev->devDriver->gp);
371 dev->devDriver = NULL;
372 }
373
374 if (dev->timer.isActive)
375 cancelTimerCallback(&dev->timer);
376
377 while (dev->endpointListStart)
378 removeEndpointFromDevice(dev, dev->endpointListStart);
379
380 while ((child = dev->childListStart)) {
381 if (child->next)
382 child->next->prev = child->prev;
383 else
384 dev->childListEnd = child->prev;
385
386 if (child->prev)
387 child->prev->next = child->next;
388 else
389 dev->childListStart = child->next;
390
391 flushPort(child);
392
393 freeDevice(child);
394 }
395 dev->ioRequest.busyFlag = 0;
396 dev->privDataField = NULL;
397 }
398 if (dev->resetFlag)
399 checkDelayedResets(dev);
400}

402void fetchConfigDescriptors(IoRequest *req)
403{
404 Endpoint *ep = req->correspEndpoint;
405 Device *dev = ep->correspDevice;
406 u16 readLen;
407
408 if ((req->resultCode == USB_RC_OK) || (dev->fetchDescriptorCounter == 0)) {
409 int fetchDesc;
410
411 u32 curDescNum = dev->fetchDescriptorCounter++;
412
413 fetchDesc = curDescNum & 1;
414 curDescNum >>= 1;
415
416 if ((curDescNum > 0) && !fetchDesc) {
417 UsbConfigDescriptor *desc = dev->staticDeviceDescEndPtr;
418 dev->staticDeviceDescEndPtr = (void *)((u8 *)(dev->staticDeviceDescEndPtr) + READ_UINT16(&desc->wTotalLength));
419 }
420
421 if (fetchDesc) {
422 UsbConfigDescriptor *desc = dev->staticDeviceDescEndPtr;
423 readLen = READ_UINT16(&desc->wTotalLength);
424 } else
425 readLen = 4;
426
427 if ((u8 *)dev->staticDeviceDescEndPtr + readLen > (u8 *)dev->staticDeviceDescPtr + usbConfig.maxStaticDescSize) {
428 dbg_printf("USBD: Device ignored, Device descriptors too large\n");
429 return; // buffer is too small, silently ignore the device
430 }
431
432 if (curDescNum < ((UsbDeviceDescriptor *)dev->staticDeviceDescPtr)->bNumConfigurations) {
433 doControlTransfer(ep, &dev->ioRequest,
434 USB_DIR_IN | USB_RECIP_DEVICE, USB_REQ_GET_DESCRIPTOR, (USB_DT_CONFIG << 8) | curDescNum, 0, readLen,
435 dev->staticDeviceDescEndPtr, fetchConfigDescriptors);
436 } else
437 connectNewDevice(dev);
438 } else
439 killDevice(dev, ep);
440}

454 } else {

457 fetchConfigDescriptors(req);
458 }
459 } else {

461 killDevice(dev, ep);

523void hubGetPortStatusCallback(IoRequest *req)
524{
525 UsbHub *dev = (UsbHub *)req->userCallbackArg;
526 Device *port;
527 if (req->resultCode == USB_RC_OK) {
528 int feature = -1;
529
530 dbg_printf("port status change: %d: %08X\n", dev->portCounter, dev->portStatusChange);
531 if (dev->portStatusChange & BIT(C_PORT_CONNECTION))
532 feature = C_PORT_CONNECTION;
533 else if (dev->portStatusChange & BIT(C_PORT_ENABLE))
534 feature = C_PORT_ENABLE;
535 else if (dev->portStatusChange & BIT(C_PORT_SUSPEND))
536 feature = C_PORT_SUSPEND;
537 else if (dev->portStatusChange & BIT(C_PORT_OVER_CURRENT))
538 feature = C_PORT_OVER_CURRENT;
539 else if (dev->portStatusChange & BIT(C_PORT_RESET))
540 feature = C_PORT_RESET;
541
542 if (feature >= 0) {
543 dev->portStatusChange &= ~BIT(feature);
544 HubControlTransfer(dev,
545 USB_DIR_OUT | USB_RT_PORT, USB_REQ_CLEAR_FEATURE, feature, dev->portCounter, 0, NULL,
546 hubGetPortStatusCallback);
547 } else {
548 port = fetchPortElemByNumber(dev->controlEp->correspDevice, dev->portCounter);
549 if (port) {
550 if (dev->portStatusChange & BIT(PORT_CONNECTION)) {
551 dbg_printf("Hub Port CCS\n");
552 if (port->deviceStatus == DEVICE_NOTCONNECTED) {
553 dbg_printf("resetting dev\n");
554 port->deviceStatus = DEVICE_CONNECTED;
555 addTimerCallback(&port->timer, (TimerCallback)hubResetDevice, port, 500);
556 return;
557 } else if (port->deviceStatus == DEVICE_RESETPENDING) {
558 if (dev->portStatusChange & BIT(PORT_ENABLE)) {
559 dbg_printf("hub port reset done, opening control EP\n");
560 port->deviceStatus = DEVICE_RESETCOMPLETE;
561 port->isLowSpeedDevice = (dev->portStatusChange >> PORT_LOW_SPEED) & 1;
562
563 if (openDeviceEndpoint(port, NULL, 0))
564 hubTimedSetFuncAddress(port);
565 else
566 dbg_printf("Can't open default control ep.\n");
567 dev->hubStatusCounter = 0;
568 }
569 }
570 } else {
571 dbg_printf("disconnected; flushing port\n");
572 flushPort(port);
573 }
574 }
575 hubStatusChangeCallback(&dev->statusIoReq);
576 }
577 } else
578 dbg_printf("HubGetPortStatusCallback res %d\n", req->resultCode);
579}

665int hubDrvConnect(int devId)
666{
667 UsbConfigDescriptor *confDesc;
668 UsbInterfaceDescriptor *intfDesc;
669 UsbEndpointDescriptor *endpDesc;
670 UsbHubDescriptor *hubDesc;
671 Device *dev;
672 UsbHub *hubDevice;
673
674 dev = fetchDeviceById(devId);
675 if (!dev)
676 return -1;
677
678 confDesc = doGetDeviceStaticDescriptor(devId, NULL, USB_DT_CONFIG);
679 if (!confDesc || (confDesc->bNumInterfaces != 1))
680 return -2;
681
682 intfDesc = doGetDeviceStaticDescriptor(devId, confDesc, USB_DT_INTERFACE);
683 if (!intfDesc || (intfDesc->bNumEndpoints != 1))
684 return -3;
685
686 endpDesc = doGetDeviceStaticDescriptor(devId, intfDesc, USB_DT_ENDPOINT);
687 if (!endpDesc)
688 return -4;
689
690 if ((endpDesc->bEndpointAddress & USB_DIR_IN) == 0)
691 return -5;
692
693 if ((endpDesc->bmAttributes & USB_ENDPOINT_XFERTYPE_MASK) != USB_ENDPOINT_XFER_INT)
694 return -6;
695
696 hubDesc = doGetDeviceStaticDescriptor(devId, endpDesc, USB_DT_HUB);
697
698 hubDevice = allocHubBuffer();
699 if (!hubDevice)
700 return -8;
701
702 dev->privDataField = hubDevice;
703 hubDevice->controlEp = dev->endpointListStart;
704
705 hubDevice->statusChangeEp = doOpenEndpoint(dev, endpDesc, 0);
706 if (!hubDevice->statusChangeEp) {
707 freeHubBuffer(hubDevice);
708 return -9;
709 }
710
711 hubDevice->controlIoReq.userCallbackArg = hubDevice;
712 hubDevice->statusIoReq.userCallbackArg = hubDevice;
713
714 if (hubDesc) {
715 u8 len = hubDesc->bLength;
716 if (len > sizeof(UsbHubDescriptor))
717 len = sizeof(UsbHubDescriptor);
718 memcpy(&hubDevice->desc, hubDesc, len);
719 } else
720 memset(&hubDevice->desc, 0, sizeof(UsbHubDescriptor));
721
722 HubControlTransfer(hubDevice,
723 USB_DIR_OUT, USB_REQ_SET_CONFIGURATION, confDesc->bConfigurationValue, 0, 0, NULL,
724 hubCheckDeviceDesc);
725
726 return 0;
727}
