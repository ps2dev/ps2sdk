21static int sceUsbdControlTransfer(int epID, int reqtyp, int req, int val, int index, int leng, void *dataptr, void *doneCB, void *arg)
22{
23 UsbDeviceRequest devreq;
24 devreq.requesttype = reqtyp;
25 devreq.request = req;
26 devreq.value = val;
27 devreq.index = index;
28 devreq.length = leng;
29
30 return sceUsbdTransferPipe(epID, dataptr, devreq.length, &devreq, doneCB, arg);
31}
