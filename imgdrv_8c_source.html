73static int imgdrv_read(iop_file_t *f, void *buf, int size)
74{
75 int i;
76 const u32 *img_ptr;
77 u32 *img_out;
78
79 img_out = (u32 *)buf;
80 img_ptr = (const u32 *)img[f->unit];
81 for (i = size; i > 0; i -= 4, img_ptr++, img_out++)
82 *img_out = *img_ptr;
83
84 return size;
85}
