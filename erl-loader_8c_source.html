50static void export_symbols() {
51 struct export_list_t * p;
52
53 for (p = export_list; p->name; p++) {
54 int i;
55 int prohibit;
56 prohibit = 0;
57 for (i = 0; prohibit_list[i]; i++) {
58 if (!(strcmp(prohibit_list[i], p->name))) {
59 prohibit = 1;
60 break;
61 }
62 }
63 if (!prohibit)
64 erl_add_global_symbol(p->name, (u32) p->pointer);
65 }
