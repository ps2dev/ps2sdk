244 static int read_sectors(void *dest, int sector, int count)
245 {
246  sceCdRMode mode;
247  int max_retries = 32;
248  int tries = 0;
249 
250  mode.trycount = max_retries;
251  mode.spindlctrl = SCECdSpinNom;
252  mode.datapattern = SCECdSecS2048;
253  mode.pad = 0;
254 
255  while (tries < max_retries)
256  {
257  /* wait until CD is ready to receive commands */
258  sceCdDiskReady(0);
259 
260  if (sceCdReadCDDA(sector, count, dest, &mode))
261  {
262  /* success! */
263  break;
264  }
265 
266  tries++;
267  }
268 
269  sceCdSync(0);
270 
271  if (tries == max_retries)
272  {
273  return -1;
274  }
275 
276  return count;
277 }

313 static void cdda_procedure(void *arg)
314 {
315  int block, sz;
316  u8 *bufptr;
317  int nsectors;
318  int intr_state;
319  int offset, last_read;
320  upsample_t up;
321  upsampler_t up44k1;
322 
323  printf("cdda_procedure started with %d, %d\n", cdda_play_start, cdda_play_end);
324 
325  if (cdda_play_start >= cdda_play_end)
326  {
327  /* nothing to play :| */
328  return;
329  }
330 
331  /* audio discs are 44100hz, 16bit, stereo */
332  up44k1 = find_upsampler(44100, 16, 2);
333 
334  cdda_pos = cdda_play_start;
335  printf("playing sectors %d - %d (%d sectors)\n", cdda_play_start, cdda_play_end, cdda_play_end - cdda_play_start);
336 
337  /* fill entire buffer before starting to play */
338  offset = 0;
339  nsectors = sizeof(cd_ringbuf) / SECTOR_SIZE;
340  printf("filling buffer with nsectors %d..\n", nsectors);
341  if (read_sectors(cd_ringbuf, cdda_pos, nsectors) != nsectors)
342  {
343  printf("failed to read %d sectors..\n", nsectors);
344  return;
345  }
346 
347  cdda_pos = cdda_pos + nsectors;
348 
349  printf("sectors read.. now setting callbacks\n");
350 
351  /* kick cd streaming */
352  sceSdSetTransCallback(SD_CORE_0, (void *)cd_transfer_complete);
353  sceSdBlockTrans(SD_CORE_0, SD_TRANS_LOOP, core0_buf, sizeof(core0_buf), 0);
354 
355  printf("callbacks kicked, starting loop\n");
356  last_read = 0;
357 
358  memset(cd_rendered_left, 0, sizeof(cd_rendered_left));
359  memset(cd_rendered_right, 0, sizeof(cd_rendered_right));
360 
361  while (cdda_pos < cdda_play_end)
362  {
363  /* wait until it's safe to transmit */
364  WaitSema(cd_transfer_sema);
365 
366  /* suspend all interrupts */
367  CpuSuspendIntr(&intr_state);
368 
369  block = 1 - (sceSdBlockTransStatus(SD_CORE_0, 0) >> 24);
370  bufptr = core0_buf + (block << 11);
371 
372  if (cd_paused == 0)
373  {
374  up.left = (short *)cd_rendered_left;
375  up.right = (short *)cd_rendered_right;
376 
377  if (offset + 1880 < sizeof(cd_ringbuf))
378  {
379  /* enough bytes in ringbuffer */
380  up.src = cd_ringbuf + offset;
381  offset = offset + up44k1(&up);
382  }
383  else
384  {
385  /* two portions */
386  sz = sizeof(cd_ringbuf) - offset;
387  wmemcpy(cd_sparebuf, cd_ringbuf + offset, sz);
388  wmemcpy(cd_sparebuf + sz, cd_ringbuf + 0, 1880 - sz);
389  offset = 1880 - sz;
390  up.src = cd_sparebuf;
391  up44k1(&up);
392  }
393 
394  /* upsample 44k1 -> 48k0 */
395  wmemcpy(bufptr + 0, cd_rendered_left + 0, 512);
396  wmemcpy(bufptr + 512, cd_rendered_right + 0, 512);
397  wmemcpy(bufptr + 1024, cd_rendered_left + 256, 512);
398  wmemcpy(bufptr + 1536, cd_rendered_right + 256, 512);
399  }
400  else
401  {
402  /* paused, send mute */
403  memset(bufptr, '\0', 2048);
404  }
405 
406  CpuResumeIntr(intr_state);
407 
408  if ((offset / SECTOR_SIZE) != last_read)
409  {
410  /* read another sector.. */
411  if (read_sectors(cd_ringbuf + (last_read * SECTOR_SIZE), cdda_pos, 1) != 1)
412  {
413  printf("failed to read 1 sector\n");
414  break;
415  }
416 
417  last_read = (offset / SECTOR_SIZE);
418  cdda_pos++;
419  }
420  }
421 
422  audsrv_stop_cd();
423 
424  /* notify cdda ended */
425  call_client_callback(AUDSRV_CDDA_CALLBACK);
426 }
