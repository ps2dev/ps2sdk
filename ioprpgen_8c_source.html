23 static int ioprpgen_membuf_writecb(void *userdata, const struct ioprpgen_ctx *ctx, const void *buf, u32 size)
24 {
25  struct ioprpgen_memwrite_ctx *memwrite_ctx;
26 
27  (void)ctx;
28  memwrite_ctx = (struct ioprpgen_memwrite_ctx *)userdata;
29  if ( (memwrite_ctx->m_write_ptr_curpos + size) > memwrite_ctx->m_write_ptr_size )
30  return 0;
31  if ( buf )
32  memcpy(&((u8 *)(memwrite_ctx->m_write_ptr))[memwrite_ctx->m_write_ptr_curpos], buf, size);
33  else
34  memset(&((u8 *)(memwrite_ctx->m_write_ptr))[memwrite_ctx->m_write_ptr_curpos], 0, size);
35  memwrite_ctx->m_write_ptr_curpos += size;
36 
37  return size;
38 }

85 u32 ioprpgen_write_ioprp(const struct ioprpgen_ctx *ctx, const struct ioprpgen_entry *entries)
86 {
87  int entry_count;
88  int i;
89  u32 romdir_size;
90  u32 extinfo_size;
91  u32 total_size;
92 
93  entry_count = 0;
94  for ( i = 0; !!entries[i].m_name; i += 1 )
95  {
96  if (
97  !strcmp(entries[i].m_name, "RESET") || !strcmp(entries[i].m_name, "ROMDIR")
98  || !strcmp(entries[i].m_name, "EXTINFO") )
99  return 0;
100  entry_count += 1;
101  }
102  if ( !ioprpgen_write_romdir_entry(ctx, "RESET", 0, 0) )
103  return 0;
104  romdir_size = 16 * (entry_count + 4);
105  if ( !ioprpgen_write_romdir_entry(ctx, "ROMDIR", 0, romdir_size) )
106  return 0;
107  total_size = romdir_size;
108  extinfo_size = 4 * entry_count;
109  if ( !ioprpgen_write_romdir_entry(ctx, "EXTINFO", 0, extinfo_size) )
110  return 0;
111  for ( i = 0; i < entry_count; i += 1 )
112  if ( !ioprpgen_write_romdir_entry(ctx, entries[i].m_name, 4, entries[i].m_data_size) )
113  return 0;
114  if ( !ioprpgen_write_romdir_entry(ctx, NULL, 0, 0) )
115  return 0;
116  {
117  u32 cur_entry_size;
118  u32 padding_size;
119  u32 cur_padding_size;
120  struct ioprp_extinfo_entry extinfo_entry;
121 
122  // These values are needed in order for UDNL to select the module
123  extinfo_entry.m_value = 0xFFFF;
124  extinfo_entry.m_extlen = 0;
125  extinfo_entry.m_type = 2;
126  cur_entry_size = 0;
127  for ( i = 0; i < entry_count; i += 1 )
128  {
129  u32 cur_value_size;
130 
131  cur_value_size = ctx->m_write_cb(ctx->m_write_cb_userdata, ctx, &extinfo_entry, sizeof(extinfo_entry));
132  if ( cur_value_size != sizeof(extinfo_entry) )
133  return 0;
134  cur_entry_size += cur_value_size;
135  }
136  if ( cur_entry_size != extinfo_size )
137  return 0;
138  total_size += cur_entry_size;
139  padding_size = ((((cur_entry_size - 1) >> 4) + 1) << 4) - cur_entry_size;
140  cur_padding_size = ctx->m_write_cb(ctx->m_write_cb_userdata, ctx, NULL, padding_size);
141  if ( cur_padding_size != padding_size )
142  return 0;
143  total_size += cur_padding_size;
144  }
145  for ( i = 0; i < entry_count; i += 1 )
146  {
147  u32 cur_entry_size;
148  u32 padding_size;
149  u32 cur_padding_size;
150 
151  cur_entry_size = ctx->m_write_cb(ctx->m_write_cb_userdata, ctx, entries[i].m_data, entries[i].m_data_size);
152  if ( cur_entry_size != entries[i].m_data_size )
153  return 0;
154  total_size += cur_entry_size;
155  padding_size = ((((cur_entry_size - 1) >> 4) + 1) << 4) - cur_entry_size;
156  cur_padding_size = ctx->m_write_cb(ctx->m_write_cb_userdata, ctx, NULL, padding_size);
157  if ( cur_padding_size != padding_size )
158  return 0;
159  total_size += cur_padding_size;
160  }
161  return total_size;
162 }
