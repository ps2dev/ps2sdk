31 static u32 LE32(u32 b) {
32  u32 t = 0x12345678;
33  if (*((unsigned char*)(&t)) == 0x78) {
34  return b;
35  } else {
36  return ((b & 0xff000000) >> 24) |
37  ((b & 0x00ff0000) >> 8 ) |
38  ((b & 0x0000ff00) << 8 ) |
39  ((b & 0x000000ff) << 24);
40  }
41 }

43 static u16 LE16(u16 b) {
44  u32 t = 0x12345678;
45  if (*((unsigned char*)(&t)) == 0x78) {
46  return b;
47  } else {
48  return ((b & 0xff00) >> 8) |
49  ((b & 0x00ff) << 8);
50  }
51 }

105  elf_header[37] = 0;

118

120  strcpy(strtab + 1, label);
121  strcat(strtab + 1, "_start");
122  strcpy(strtab + 1 + l_size + 7, label);
123  strcat(strtab + 1 + l_size + 7, "_end");
124  if (have_size) {
125  strtab_size = (l_size * 3 + 1 + 7 + 5 + 6);
126  strcpy(strtab + 1 + l_size + 7 + l_size + 5, label);
127  strcat(strtab + 1 + l_size + 7 + l_size + 5, "_size");
128  } else {
129  strtab_size = (l_size * 2 + 1 + 7 + 5);
130  }

132  // section 0 (NULL)
133  section.sh_name = section.sh_type = section.sh_flags = section.sh_addr = section.sh_offset = section.sh_size =
134  section.sh_link = section.sh_info = section.sh_addralign = section.sh_entsize = 0;
135 
136  fwrite(&section, 1, sizeof(section), dest);
137 
138  // section 1 (.shstrtab)
139  if(put_payload_in_SDATA || put_labels_in_SDATA){
140  pshstrtab_ptr=shstrtab_sdata;
141  shrtabSectionNameLength=sizeof(shstrtab_sdata);
142  }else{
143  pshstrtab_ptr=shstrtab;

147  section.sh_type = LE32(3); // STRTAB
148  section.sh_flags = 0;
149  section.sh_addr = 0;

151  section.sh_size = LE32(shrtabSectionNameLength);
152  section.sh_link = 0;
153  section.sh_info = 0;
154  section.sh_addralign = LE32(1);
155  section.sh_entsize = 0;

163  section.sh_flags = 0;
164  section.sh_addr = 0;
165  section.sh_offset = LE32(40 * NumSections + 0x34 + shrtabSectionNameLength);
166  section.sh_size = LE32(have_size ? 0x30 : 0x20);
167  section.sh_link = LE32(3);
168  section.sh_info = 0;
169  section.sh_addralign = LE32(4);
170  section.sh_entsize = LE32(0x10);
171 
172  fwrite(&section, 1, sizeof(section), dest);
173 
174  // section 3 (.strtab)
175  section.sh_name = LE32(19);
176  section.sh_type = LE32(3); // STRTAB
177  section.sh_flags = 0;
178  section.sh_addr = 0;
179  section.sh_offset = LE32(40 * NumSections + 0x34 + shrtabSectionNameLength + (have_size ? 0x30 : 0x20));
180  section.sh_size = LE32(strtab_size);
181  section.sh_link = 0;
182  section.sh_info = 0;
183  section.sh_addralign = LE32(1);
184  section.sh_entsize = 0;

189  // section 4 (.sdata)
190  section.sh_name = LE32(27);
191  section.sh_type = LE32(1); // PROGBITS
192  section.sh_flags = LE32(0x10000003); // SHF_MIPS_GPREL + write + alloc
193  section.sh_addr = 0;
194  section.sh_offset = LE32(40 * NumSections + 0x34 + shrtabSectionNameLength + (have_size ? 0x30 : 0x20) + strtab_size);

196  section.sh_link = 0;
197  section.sh_addralign = LE32(alignment);
198  section.sh_entsize = 0;
199

201  }

204  section.sh_name = LE32((put_payload_in_SDATA||put_labels_in_SDATA)?35:27);
205  section.sh_type = LE32(1); // PROGBITS
206  section.sh_flags = LE32(3); // Write + Alloc
207  section.sh_addr = 0;
208  if(!put_payload_in_SDATA || !put_labels_in_SDATA){
209  section.sh_offset = LE32(40 * NumSections + 0x34 + shrtabSectionNameLength + (have_size ? 0x30 : 0x20) + strtab_size + ((put_payload_in_SDATA?size:0) + (put_labels_in_SDATA?have_size*16:0)));
210  section.sh_size = LE32(((put_payload_in_SDATA?0:size) + (put_labels_in_SDATA?0:have_size*16)));
211  }
212  else{
213  section.sh_offset = 0;
214  section.sh_size = 0;
215  }
216  section.sh_link = 0;
217  section.sh_addralign = LE32(alignment);
218  section.sh_entsize = 0;
219 
220  fwrite(&section, 1, sizeof(section), dest);
221 
222  fwrite(pshstrtab_ptr, 1, shrtabSectionNameLength, dest);
223 
224  symbol.st_name = LE32(1);
225  symbol.st_value = LE32((have_size&&put_payload_in_SDATA==put_labels_in_SDATA)?16:0); //If the size label and the payload exist in the same section, the size label will exist before the payload.
226  symbol.st_shndx = LE16((put_labels_in_SDATA&&!put_payload_in_SDATA)?5:4);
227  symbol.st_size = LE32(size);
228  symbol.st_info = 0x11;
229  symbol.st_other = 0;
230  fwrite(&symbol, 1, sizeof(symbol), dest);
231 
232  symbol.st_name = LE32(l_size + 1 + 7);
233  symbol.st_value = LE32(size + have_size * 16);
234  symbol.st_size = 0;
235  symbol.st_info = 0x11;
236  symbol.st_other = 0;
237  symbol.st_shndx = LE16(4);
238  fwrite(&symbol, 1, sizeof(symbol), dest);
239 
240  if (have_size) {
241  symbol.st_name = LE32(l_size * 2 + 1 + 7 + 5);
242  symbol.st_value = 0;
243  symbol.st_size = LE32(4);
244  symbol.st_info = 0x11;
245  symbol.st_other = 0;
246  symbol.st_shndx = LE16(4);
247  fwrite(&symbol, 1, sizeof(symbol), dest);
248  }
249 
250  fwrite(strtab, 1, strtab_size, dest);
251 
252  if (have_size) {
253  data_size[0] = LE32(size);
254  data_size[1] = 0;
255  data_size[2] = 0;
256  data_size[3] = 0;
257  fwrite(data_size, 4, 4, dest);
258  }
259  fwrite(source, 1, size, dest);
260 }

286  for (i = 1; i < argc; i++) {

288  switch (argv[i][1]) {
289  case 'a':
290  i++;
291  if (!argv[i]) {
292  usage();
293  printf("-a requires an argument.\n");
294  return 1;
295  }
296  if (argv[i][0] == '-') {
297  usage();
298  printf("-a requires an argument.\n");

300  }

303  printf("Error: alignment must be a power of 2.\n");

310  case 'i':

312  break;
313  case 'G':
314  SDATA_size_limit = strtoul(&argv[i][2], NULL, 0);
315  break;
316  case 'b':
317  i++;
318  if (!argv[i]) {
319  usage();
320  printf("-b requires an argument.\n");
321  return 1;
322  }
323  if (argv[i][0] == '-') {
324  usage();
325  printf("-b requires an argument.\n");
326  return 1;
327  }
328  start = atoi(argv[i]);
329  break;
330  case 'e':
331  i++;
332  if (!argv[i]) {
333  usage();
334  printf("-e requires an argument.\n");
335  return 1;
336  }
337  if (argv[i][0] == '-') {
338  usage();
339  printf("-e requires an argument.\n");
340  return 1;
341  }
342  end = atoi(argv[i]);
343  break;
344  case 's':
345  i++;
346  if (!argv[i]) {
347  usage();
348  printf("-s requires an argument.\n");
349  return 1;
350  }
351  if (argv[i][0] == '-') {
352  usage();
353  printf("-s requires an argument.\n");
354  return 1;
355  }
356  size = atoi(argv[i]);
357  break;
358  default:
359  usage();
360  printf("Unknow option: %s.\n", argv[i]);
361  return 1;
362  }
363  } else {
364  if (!f_source) {
365  f_source = argv[i];
366  } else if (!f_dest) {
367  f_dest = argv[i];
368  } else if (!f_label) {
369  f_label = argv[i];
370  } else {
371  usage();
372  printf("Too many arguments.\n");
373  return 1;
374  }
375  }
376  }
377 
378  if (!f_source || !f_dest || !f_label) {
379  usage();
380  printf("Not enough arguments.\n");
381  return 1;
382  }
383 
384  if (!(source = fopen(f_source, "rb"))) {
385  printf("Error opening %s for reading.\n", f_source);
386  return 1;
387  }
388 
389  fseek(source, 0, SEEK_END);
390  fd_size = ftell(source);
391  fseek(source, start, SEEK_SET);
392 
393  if (fd_size < end)
394  end = fd_size;
395 
396  if (end < (size - start))
397  size = end - start;
398 
399  if(SDATA_size_limit>0){
400  put_payload_in_SDATA=(SDATA_size_limit>=size)?1:0;
401  put_labels_in_SDATA=(have_size && SDATA_size_limit>=16)?1:0;
402  }
403  else{
404  put_payload_in_SDATA = put_labels_in_SDATA = 0;
405  }
406 
407  buffer = malloc(size);
408  if (buffer == NULL) {
409  printf("Failed to allocate memory.\n");
410  return 1;
411  }
412 
413  if (fread(buffer, 1, size, source) != size) {
414  printf("Failed to read file.\n");
415  return 1;
416  }
417  fclose(source);
418 
419  if (!(dest = fopen(f_dest, "wb+"))) {
420  printf("Failed to open/create %s.\n", f_dest);
421  return 1;
422  }
423 
424  create_elf(dest, buffer, size, f_label);
425 
426  fclose(dest);
427  free(buffer);
428 
429  return 0;
