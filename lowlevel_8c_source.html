22static s8 twh(s16 val)
23{
24 s8 res;
25
26 asm volatile ("plzcw %0, %1\n": "=r" (res) : "r" (val));
27 res = 31 - (res + 1);
28 if(val > res)
29 res++;
30
31 return res;
32}
