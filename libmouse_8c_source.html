41int PS2MouseInit(void)
42
43{
44 if(mouse_init)
45 {
46 printf("PS2Mouse Library already initialised\n");
47 return 0;
48 }
49
50 mouseif.server = NULL;
51
52 do {
53 if (sceSifBindRpc(&mouseif, PS2MOUSE_BIND_RPC_ID, 0) < 0) {
54 return -1;
55 }
56 nopdelay();
57 } while(!mouseif.server);
58
59 mouse_init = 1;
60
61 return 1;
62}
