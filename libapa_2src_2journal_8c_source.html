59 int apaJournalRestore(s32 device)
60 { // copies the journal from the HDD and erases the original.
61  int i, ret;
62  u32 sector;
63  apa_cache_t *clink;
64 
65  APA_PRINTF(APA_DRV_NAME": checking log...\n");
66  ret = ata_device_sector_io(device, &journalBuf, APA_SECTOR_APAL, sizeof(apa_journal_t)/512, ATA_DIR_READ) == 0 ? 0 : -EIO;
67  if((ret == 0) && (journalBuf.magic == APAL_MAGIC))
68  {
69  if(journalBuf.num == 0)
70  return 0;
71 
72  clink=apaCacheAlloc();
73  for(i=0, sector=APA_SECTOR_APAL_HEADERS;i<journalBuf.num;i++, sector+=2)
74  {
75  ret = (ata_device_sector_io(device, clink->header, sector, 2, ATA_DIR_READ) == 0) ? 0 : -EIO;
76  if(ret != 0)
77  break;
78 
79  ret = (ata_device_sector_io(device, clink->header, journalBuf.sectors[i], 2, ATA_DIR_WRITE) == 0) ? 0 : -EIO;
80  if(ret != 0)
81  break;
