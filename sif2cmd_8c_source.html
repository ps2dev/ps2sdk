115 void SIF2_send_cmd(u32 cid, void *extra, int extra_size)
116 {
117  SIF2_CmdPkt *cmd = (SIF2_CmdPkt *) _sif2_dma_cmd_buf;
118 
119  cmd->cid = cid;
120  cmd->size = sizeof(SIF2_CmdPkt);
121  cmd->extra = 0;
122  cmd->extra_size = 0;
123 
124  if(extra)
125  {
126  if(extra_size <= (sizeof(_sif2_dma_cmd_buf) - sizeof(SIF2_CmdPkt)))
127  {
128  memcpy((u32 *) (((u32) &_sif2_dma_cmd_buf) + sizeof(SIF2_CmdPkt)), extra, extra_size);
129  cmd->size += extra_size;
130  }
131  else
132  {
133  cmd->extra = (u32) extra;
134  cmd->extra_size = extra_size;
135  }
136  }
137 
138 #ifdef _EE
139  WriteBackDCache(_sif2_dma_cmd_buf, sizeof(_sif2_dma_cmd_buf));
140 #endif
141 
142  // set up the DMA transfer to send the cmd packet to remote.
143  SIF2_set_dma((u32) cmd, sizeof(_sif2_dma_cmd_buf), PS2_DMA_FROM_MEM);
144 
145  // interrupt the remote CPU with "EXEC".
146  SBUS_interrupt_remote(SBUS_IRQ_EXEC);
147 
148  // wait for transfer to complete.
149  SIF2_sync_dma();
150 
151  if(cmd->extra)
152  {
153 #ifdef _EE
154  WriteBackDCache((void *) cmd->extra, cmd->extra_size);
155 #endif
156 
157  // set up the DMA transfer to send the extra data to remote.
158  SIF2_set_dma((u32) cmd->extra, cmd->extra_size, PS2_DMA_FROM_MEM);
159 
160  // wait for transfer to complete.
161  SIF2_sync_dma();
162  }
163 }
