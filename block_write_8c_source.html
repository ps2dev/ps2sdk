53 
54  if (pfsCacheIsFull())
55  return -ENOMEM;
56 
57  // create "indirect segment descriptor" if necessary
58  if (pfsFixIndex(clink->u.inode->number_data) == 0)

67  if (result<0)

73  clink2=pfsCacheGetData(pfsMount, bi.subpart, bi.number << pfsMount->inode_scale,
74  PFS_CACHE_FLAG_SEGI | PFS_CACHE_FLAG_NOLOAD, &result);
75  memset(clink2->u.inode, 0, sizeof(pfs_inode_t));
76  clink2->u.inode->magic=PFS_SEGI_MAGIC;

78  memcpy(&clink2->u.inode->inode_block, &clink->u.inode->inode_block, sizeof(pfs_blockinfo_t));
