73/*
74 * Convert a string to an unsigned long integer.
75 *
76 * Ignores `locale' stuff. Assumes that the upper and lower case
77 * alphabets and digits are each contiguous.
78 */
79unsigned long strtoul(const char* nptr, char** endptr, int base)
80{
81 const char* s = nptr;
82 unsigned long acc;
83 int c;
84 unsigned long cutoff;
85 int neg = 0;
86 int any;
87 int cutlim;
88
89 /*
90 * See strtol for comments as to the logic used.
91 */
92 do
93 {
94 c = *s++;
95 } while(isspace(c));
96
97 if(c == '-')
98 {
99 neg = 1;
100 c = *s++;
101 }
102 else if(c == '+')
103 {
104 c = *s++;
105 }
106
107 if((base == 0 || base == 16) && c == '0' && (*s == 'x' || *s == 'X'))
108 {
109 c = s[1];
110 s += 2;
111 base = 16;
112 }
113 else if((base == 0 || base == 2) && c == '0' && (*s == 'b' || *s == 'B'))
114 {
115 c = s[1];
116 s += 2;
117 base = 2;
118 }
119
120 if(base == 0)
121 {
122 base = c == '0' ? 8 : 10;
123 }
124
125 cutoff = (unsigned long)ULONG_MAX / (unsigned long)base;
126 cutlim = (int)((unsigned long)ULONG_MAX % (unsigned long)base);
127
128 for(acc = 0, any = 0;; c = *s++)
129 {
130 if(isdigit(c))
131 {
132 c -= '0';
133 }
134 else if(isalpha(c))
135 {
136 c -= isupper(c) ? 'A' - 10 : 'a' - 10;
137 }
138 else
139 {
140 break;
141 }
142 if(c >= base)
143 {
144 break;
145 }
146 if(any < 0 || acc > cutoff || (acc == cutoff && c > cutlim))
147 {
148 any = -1;
149 }
150 else
151 {
152 any = 1;
153 acc *= (unsigned long)base;
154 acc += (unsigned long)c;
155 }
156 }
157
158 if(any < 0)
159 {
160 acc = ULONG_MAX;
161 // errno = ERANGE;
162 }
163 else if(neg)
164 {
165 acc = -acc;
166 }
167
168 if(endptr != 0)
169 {
170 *endptr = (char*)(uintptr_t)(any ? s - 1 : nptr);
171 }
172
173 return (acc);
174}
