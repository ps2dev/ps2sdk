61 SMapLowLevelOutput(NetIF *pNetIF, PBuf *pOutput)
62 {
63  err_t result;
64  struct pbuf *pbuf;
65 
66 #if USE_GP_REGISTER
67  void *OldGP;
68 
69  OldGP = SetModuleGP();
70 #endif
71 
72  (void)pNetIF;
73 
74  result = ERR_OK;
75  if (pOutput->tot_len > pOutput->len) {
76  pbuf_ref(pOutput); // Increment reference count because LWIP must free the PBUF, not the driver!
77  if ((pbuf = pbuf_coalesce(pOutput, PBUF_RAW)) != pOutput) { // No need to increase reference count because pbuf_coalesce() does it.
78  EnQTxPacket(pbuf);
79  SMAPXmit();
80  } else
81  result = ERR_MEM;
82  } else {
83  pbuf_ref(pOutput); // This will be freed later.
84  EnQTxPacket(pOutput);
85  SMAPXmit();
86  }

89  SetGP(OldGP);
90 #endif
91 
92  return result;
93 }

142 static void EnQTxPacket(struct pbuf *tx)
143 {
144  int OldState;
145 
146  CpuSuspendIntr(&OldState);
147 
148  if (TxHead != NULL)
149  TxHead->next = tx;
150 
151  TxHead = tx;
152  tx->next = NULL;
153 
154  if (TxTail == NULL) // Queue empty
155  TxTail = TxHead;

157  CpuResumeIntr(OldState);

173 void SMapTxPacketDeQ(void)
174 {
175  struct pbuf *toFree;
176  int OldState;
177 
178  toFree = NULL;
179 
180  CpuSuspendIntr(&OldState);
181  if (TxTail != NULL) {
182  toFree = TxTail;
183 
184  if (TxTail == TxHead) {
185  // Last in queue.
186  TxTail = NULL;
187  TxHead = NULL;
188  } else {
189  TxTail = TxTail->next;
190  }
191  }
192  CpuResumeIntr(OldState);
193 
194  if (toFree != NULL) {
195  toFree->next = NULL;
196  pbuf_free(toFree);
