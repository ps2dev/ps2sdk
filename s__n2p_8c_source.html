40u16 _spu_note2pitch(u16 cen_note_high, u16 cen_note_low, u16 note_high, u16 note_low)
41{
42 u16 max_lo;
43 int calc_note;
44 u16 max_lo_idx;
45 int calc_note_div12;
46 int calc_note_div12_min12;
47 int calc_note_mod12;
48 int calc_note_mod12_idx;
49
50 max_lo = note_low + cen_note_low;
51 calc_note = (s16)(note_high + (max_lo >> 7) - cen_note_high);
52 max_lo_idx = max_lo & 0x7F;
53 calc_note_div12 = calc_note / 12;
54 calc_note_div12_min12 = calc_note / 12 - 2;
55 calc_note_mod12 = calc_note % 12;
56 calc_note_mod12_idx = calc_note_mod12;
57 if ( (calc_note_mod12 & 0x8000) != 0 )
58 {
59 calc_note_mod12_idx = calc_note_mod12 + 12;
60 calc_note_div12_min12 = calc_note_div12 - 3;
61 }
62 if ( (calc_note_div12_min12 & 0x8000u) == 0 )
63 return 0x3FFF;
64 return (unsigned int)(((_spu_NTable[calc_note_mod12_idx] * (u16)_spu_FTable[max_lo_idx]) >> 16)
65 + (1 << (-(s16)calc_note_div12_min12 - 1)))
66 >> -(s16)calc_note_div12_min12;
67}

69int _spu_pitch2note(s16 note_high, s16 note_low, u16 pitch)
70{
71 s16 v3;
72 u16 v4;
73 s16 v5;
74 int v6;
75 int v8;
76 int v9;
77 int v11;
78
79 v3 = 0;
80 v4 = 0;
81 v5 = 0;
82 if ( pitch >= 0x4000u )
83 pitch = 0x3FFF;
84 for ( v6 = 0; v6 < 14; v6 += 1 )
85 {
86 if ( (((int)pitch >> v6) & 1) != 0 )
87 v5 = v6;
88 }
89 v8 = pitch << (15 - v5);
90 for ( v9 = 11; v9 >= 0; v9 -= 1 )
91 {
92 if ( (u16)v8 >= (unsigned int)(u16)_spu_NTable[v9] )
93 {
94 v4 = v9;
95 break;
96 }
97 }
98 if ( !_spu_NTable[v4] )
99 __builtin_trap();
100 for ( v11 = 127; v11 >= 0; v11 -= 1 )
101 {
102 if ( (u16)(((u16)v8 << 15) / (unsigned int)(u16)_spu_NTable[v4]) >= (unsigned int)(u16)_spu_FTable[v11] )
103 {
104 v3 = v11;
105 break;
106 }
107 }
108 return ((u16)(v4 + note_high + 12 * (v5 - 12) + ((u16)(note_low + v3 + 1) >> 7)) << 8) | ((note_low + v3 + 1) & 0x7E);
109}
