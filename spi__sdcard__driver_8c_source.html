53 static uint8_t _send_command(uint8_t cmd, uint32_t arg) {
54  uint32_t i;
55  uint8_t response = 0xFF;
56 
57  /* Dummy byte and chip enable */
58  _io->wr_rd_byte(DUMMY_BYTE);
59  _io->select();
60 
61  uint8_t packet[] = {cmd | 0x40, arg >> 24, arg >> 16, arg >> 8, arg, 0};
62  packet[CMD_CRC_OFFSET] = CRC7_SHIFT_MASK(crc7(packet, CMD_CRC_OFFSET));
63 
64  _io->write(packet, sizeof(packet));
65 
66  /* Discard stuff byte for CMD12 */
67  if (cmd == CMD12) {
68  _io->wr_rd_byte(DUMMY_BYTE);
69  }
70 
71  /* Wait response, quit till timeout */
72  for (i = 0; i < CMD_WAIT_RESP_TIMEOUT; i++) {
73  response = _io->wr_rd_byte(DUMMY_BYTE);
74 
75  if (response != 0xFF) {
76  break;
77  }
78  }
79 
80  /* Chip disable and dummy byte */
81  _io->relese();
82  _io->wr_rd_byte(DUMMY_BYTE);
83 
84  return response;
85 }

206  uint8_t buff[4];
207  response = _send_command_recv_response(CMD8, 0x1AA, buff, sizeof(buff));
208 
209  if (response == SPISD_R1_IDLE_FLAG) {
210 
211  /* Check voltage range be 2.7-3.6V */
212  if (buff[2] == 0x01 && buff[3] == 0xAA) {
213 
214  for (i = 0; i < 0xFFF; i++) {
215  response = _send_command(CMD55, 0); /* should be return 0x01 */

232  }
233

236

244

261

272  }
273

277  response = _send_command(CMD1, 0);
278 
279  if (response == 0x00) {
280  break;
281  }
282  }

287  return SPISD_RESULT_TIMEOUT;

294  }

514  uint32_t i, j;
515  /* if ver = SD2.0 HC, sector need <<9 */
516  if (_card_type != CARD_TYPE_SDV2HC) {

518  }
519

521  if (_card_type != CARD_TYPE_MMC) {
522  _send_command(CMD55, 0);
523  _send_command(ACMD23, num_sectors);

527  return SPISD_RESULT_ERROR;

531  _io->wr_rd_byte(DUMMY_BYTE);
532  _io->wr_rd_byte(DUMMY_BYTE);

534

536

538  _io->wr_rd_byte(0xFC);
539

541

561  return SPISD_RESULT_ERROR;
562  }
563  }
564  }

567  if (_io->wr_rd_byte(0xFD) != 0x00) {
568 
569  _io->wr_rd_byte(DUMMY_BYTE);
570  _io->wr_rd_byte(DUMMY_BYTE);

575  _io->relese();
576  _io->wr_rd_byte(DUMMY_BYTE);
577

579  if (_io->wr_rd_byte(DUMMY_BYTE) == 0xFF) {
580  return SPISD_RESULT_OK;
581  }
582  }
583  }
584  }
