20int _start(int argc, char *argv[])
21{
22 int *BootMode;
23 int v3;
24 iop_thread_t v4;
25
26 FlushDcache();
27 BootMode = QueryBootMode(3);
28 if (BootMode) {
29 int v1;
30
31 v1 = BootMode[1];
32 if ((v1 & 1) != 0) {
33 printf(" No SIF service(reboot)\n");
34 return 1;
35 }
36 if ((v1 & 2) != 0) {
37 printf(" No Reboot by EE service\n");
38 return 1;
39 }
40 }
41 CpuEnableIntr();
42 v4.attr = 0x2000000;
43 v4.thread = reboot_handler_thread;
44 v4.priority = 10;
45 v4.stacksize = 2048;
46 v4.option = 0;
47 v3 = CreateThread(&v4);
48 if (v3 <= 0) {
49 return 1;
50 }
51 StartThread(v3, 0);
52 return 0;
53}
