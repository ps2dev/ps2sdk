33 int SBUS_interrupt_remote(int irq)
34 {
35  // wait for remote to clear "irq" bit
36  while(SBUS_get_reg(PS2_SBUS_LR_FLAG) & (1 << irq));
37 
38  // set the SBUS "irq" bit for remote interrupt.
39  SBUS_set_reg(PS2_SBUS_LR_FLAG, 1 << irq);
40 
41  // cause SBUS interrupt on remote.
42  *R_IOP_IRQ_CTRL |= (1 << PS2_IRQ_SBUS);
43 
44  // clear the SBUS interrupt on remote.
45  *R_IOP_IRQ_CTRL &= ~(1 << PS2_IRQ_SBUS);
46 
47  return(irq);
48 }
