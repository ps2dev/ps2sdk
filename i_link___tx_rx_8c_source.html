31 static int iLinkSendData(struct TransactionContextData *trContext, unsigned int *header, unsigned int HeaderLength, unsigned int *payload, unsigned int PayloadLength){
32  unsigned int i, *data;
33  u32 FlagBits;
34 
35  WaitSema(UBUFTxSema);
36  ClearEventFlag(IntrEventFlag, ~(iLinkEventDataSent|iLinkEventDataReceived|iLinkEventError));
37 
38  ILINKRegisterBase->ubufTransmitClear=0;
39  data=header;
40 
41  /* Transmit the header... */
42  for(i=0; i<HeaderLength; i++){
43  if((trContext!=NULL)&&(trContext->GenerationNumber!=GenerationNumber)) return(-1021);
44 
45  ILINKRegisterBase->ubufTransmitNext=*data;
46  data++;
47  }
48 
49  /* Now transmit the payload. */
50  if(PayloadLength>0){
51  data=payload;
52  for(i=0; i<PayloadLength; i++){
53  if((trContext!=NULL)&&(trContext->GenerationNumber!=GenerationNumber)) return(-1021);
54  ILINKRegisterBase->ubufTransmitNext=*data;
55  data++;
56  }
57  }
58 
59  ILINKRegisterBase->ubufTransmitLast=*(data-1);
60 
61  DEBUG_PRINTF("Transmitting payload...\n");
62  WaitEventFlag(IntrEventFlag, iLinkEventDataSent|iLinkEventError, WEF_OR, &FlagBits);
63 
64  if(FlagBits&iLinkEventError){
65  DEBUG_PRINTF("iLinkman: TX Error.\n");
66  }
67 
68  SignalSema(UBUFTxSema);
69 
70  return((FlagBits&iLinkEventError)?-1:PayloadLength);
71 }
