149 static void *NETMAN_rpc_handler(int fno, void *buffer, int size)
150 {
151  static int ResultValue;
152  void *result;
153 
154  switch(fno)
155  {
156  case NETMAN_IOP_RPC_FUNC_INIT:
157  ResultValue=NetManInitRPCClient();
158  result=&ResultValue;
159  break;
160  case NETMAN_IOP_RPC_FUNC_DEINIT:
161  unregisterEENetworkStack();
162  NetManDeinitRPCClient();
163  result=NULL;
164  break;
165  case NETMAN_IOP_RPC_FUNC_REG_NETWORK_STACK:
166  result = &SifRpcTxBuffer;
167 
168  if(TxFrameBuffer == NULL) TxFrameBuffer = malloc(NETMAN_MAX_FRAME_SIZE * NETMAN_RPC_BLOCK_SIZE);
169  if(FrameBufferStatus == NULL) FrameBufferStatus = malloc(sizeof(struct NetManBD) * NETMAN_RPC_BLOCK_SIZE);
170 
171  if(TxFrameBuffer != NULL && FrameBufferStatus != NULL)
172  {
173  EEFrameBufferStatus = ((struct NetManRegNetworkStack*)buffer)->FrameBufferStatus;
174  memset(FrameBufferStatus, 0, sizeof(struct NetManBD) * NETMAN_RPC_BLOCK_SIZE);
175  memset(TxFrameBuffer, 0, NETMAN_MAX_FRAME_SIZE * NETMAN_RPC_BLOCK_SIZE);
176  SifRpcTxBuffer.RegNetworkStackResult.FrameBuffer = TxFrameBuffer;
177  IOPFrameBufferRdPtr = 0;
178  ResultValue = NetManRegisterNetworkStack(&RpcStack);
179 
180  if(ResultValue == 0)
181  sceSifAddCmdHandler(NETMAN_SIFCMD_ID, &HandleRxEvent, NULL);
182  }else{
183  ResultValue = -ENOMEM;
184  }
185 
186  IsRpcStackInitialized=1;
187  SifRpcTxBuffer.RegNetworkStackResult.result = ResultValue;
188  SifRpcTxBuffer.RegNetworkStackResult.FrameBufferStatus = FrameBufferStatus;
189  break;
190  case NETMAN_IOP_RPC_FUNC_UNREG_NETWORK_STACK:
191  unregisterEENetworkStack();
192  result=NULL;
193  break;
194  case NETMAN_IOP_RPC_FUNC_IOCTL:
195  SifRpcTxBuffer.IoctlResult.result=NetManIoctl(((struct NetManIoctl*)buffer)->command, ((struct NetManIoctl*)buffer)->args, ((struct NetManIoctl*)buffer)->args_len, SifRpcTxBuffer.IoctlResult.output, ((struct NetManIoctl*)buffer)->length);
196  result=&SifRpcTxBuffer;
197  break;
198  case NETMAN_IOP_RPC_FUNC_SET_MAIN_NETIF:
199  SifRpcTxBuffer.result=NetManSetMainIF(buffer);
200  result=&SifRpcTxBuffer;
201  break;
202  case NETMAN_IOP_RPC_FUNC_QUERY_MAIN_NETIF:
203  SifRpcTxBuffer.QueryMainNetIFResult.result=NetManQueryMainIF(SifRpcTxBuffer.QueryMainNetIFResult.name);
204  result=&SifRpcTxBuffer;
205  break;
206  case NETMAN_IOP_RPC_FUNC_SET_LINK_MODE:
207  ResultValue=NetManSetLinkMode(*(s32*)buffer);
208  result=&ResultValue;
209  break;
210  default:
211  printf("NETMAN [IOP]: Unrecognized command: 0x%x\n", fno);
212  ResultValue=-1;
213  result=&ResultValue;
214  }
215 
216  return result;
217 }
