65 //---------------------------------------------------------------------------
66 /* search cache records for the sector number stored in cache */
67 int getIndexRead(cache_set* cache, unsigned int sector) {
68  int i;
69  int index =-1;
70 
71  for (i = 0; i < CACHE_SIZE; i++) {
72  if (sector >= cache->rec[i].sector && sector < (cache->rec[i].sector + cache->indexLimit)) {
73  if (cache->rec[i].tax < 0) cache->rec[i].tax = 0;
74  cache->rec[i].tax +=2;
75  index = i;
76  }
77  cache->rec[i].tax--; //apply tax penalty
78  }
79  if (index < 0)
80  return index;
81  else
82  return ((index * cache->indexLimit) + (sector - cache->rec[index].sector));
83 }

85 //---------------------------------------------------------------------------
86 /* select the best record where to store new sector */
87 int getIndexWrite(cache_set* cache, unsigned int sector) {
88  int i, ret;
89  int minTax = 0x0FFFFFFF;
90  int index = 0;
91 
92  for (i = 0; i < CACHE_SIZE; i++) {
93  if (cache->rec[i].tax < minTax) {
94  index = i;
95  minTax = cache->rec[i].tax;
96  }
97  }
98 
99  //this sector is dirty - we need to flush it first
100  if (cache->rec[index].writeDirty) {
101  XPRINTF("scache: getIndexWrite: sector is dirty : %d index=%d \n", cache->rec[index].sector, index);
102  ret = WRITE_SECTOR(cache->dev, cache->rec[index].sector, cache->sectorBuf + (index * BLOCK_SIZE), BLOCK_SIZE/cache->dev->sectorSize);
103  cache->rec[index].writeDirty = 0;
104  //TODO - error handling
105  if (ret < 0) {
106  printf("scache: ERROR writing sector to disk! sector=%d\n", sector);
107  }
108 
109  }
110  cache->rec[index].tax +=2;
111  cache->rec[index].sector = sector;
112 
113  return index * cache->indexLimit;
114 }

136  for (i = 0; i < CACHE_SIZE; i++) {

139  ret = WRITE_SECTOR(cache->dev, cache->rec[i].sector, cache->sectorBuf + (i * BLOCK_SIZE), BLOCK_SIZE);
140  cache->rec[i].writeDirty = 0;
141  //TODO - error handling
142  if (ret < 0) {
143  printf("scache: ERROR writing sector to disk! sector=%d\n", cache->rec[i].sector);
144  return ret;
145  }
146  counter ++;
147  }
148  }
149  cache->writeFlag = 0;
150  return counter;
151 }

155  int index; //index is given in single sectors not octal sectors
156  int ret;
157  unsigned int alignedSector;
158

161  printf("cache: devId cache not created = %i \n", cache->dev->nodeID);
162  return -1;
163  }
164 
165  cache->cacheAccess ++;
166  index = getIndexRead(cache, sector);
167  XPRINTF("cache: indexRead=%i \n", index);
168  if (index >= 0) { //sector found in cache
169  cache->cacheHits ++;
170  *buf = cache->sectorBuf + (index * cache->sectorSize);
171  XPRINTF("cache: hit and done reading sector \n");
172 
173  return cache->sectorSize;
174  }
175 
176  //compute alignedSector - to prevent storage of duplicit sectors in slots
177  alignedSector = (sector/cache->indexLimit)*cache->indexLimit;
178  index = getIndexWrite(cache, alignedSector);
179  XPRINTF("cache: indexWrite=%i slot=%d alignedSector=%i\n", index, index / cache->indexLimit, alignedSector);
180  ret = READ_SECTOR(cache->dev, alignedSector, cache->sectorBuf + (index * cache->sectorSize), BLOCK_SIZE/cache->dev->sectorSize);
181 
182  if (ret < 0) {
183  printf("scache: ERROR reading sector from disk! sector=%d\n", alignedSector);
184  return ret;
185  }
186  *buf = cache->sectorBuf + (index * cache->sectorSize) + ((sector%cache->indexLimit) * cache->sectorSize);
187  XPRINTF("cache: done reading physical sector \n");
188

258 cache_set* scache_init(struct SBP2Device* dev, int sectSize)
259 {
260  cache_set* cache;
261  XPRINTF("cache: init devId = %i sectSize = %i \n", dev->nodeID, sectSize);
262 
263  cache = malloc(sizeof(cache_set));
264  if (cache == NULL) {
265  printf("scache init! Sector cache: can't alloate cache!\n");
266  return NULL;
267  }
268 
269  XPRINTF("scache init! \n");
270  cache->dev = dev;
271 
272  XPRINTF("sectorSize: 0x%x\n", cache->sectorSize);
273  cache->sectorBuf = (unsigned char*) malloc(BLOCK_SIZE * CACHE_SIZE);
274  if (cache->sectorBuf == NULL) {
275  printf("Sector cache: can't alloate memory of size:%d \n", BLOCK_SIZE * CACHE_SIZE);
276  free(cache);
277  return NULL;
278  }
279  XPRINTF("Sector cache: allocated memory at:%p of size:%d \n", cache->sectorBuf, BLOCK_SIZE * CACHE_SIZE);
280 
281  //added by Hermes
282  cache->sectorSize = sectSize;
283  cache->indexLimit = BLOCK_SIZE/cache->sectorSize; //number of sectors per 1 cache slot
284  cache->cacheAccess = 0;
285  cache->cacheHits = 0;
286  initRecords(cache);
287  return cache;
288 }
