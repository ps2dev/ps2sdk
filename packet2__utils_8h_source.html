138 static inline u32 packet2_utils_get_packet_size_for_program(u32 *start, u32 *end)
139 {
140 // Count instructions
141 u32 count = (end - start) / 2;
142 if (count & 1)
143 count++;
144 return (count >> 8) + 1;
145 }
