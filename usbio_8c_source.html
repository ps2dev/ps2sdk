22void removeEndpointFromQueue(Endpoint *ep)
23{
24 if (!ep->inTdQueue)
25 return;
26
27 if (ep->busyNext)
28 ep->busyNext->busyPrev = ep->busyPrev;
29 else
30 memPool.tdQueueEnd[ep->inTdQueue - 1] = ep->busyPrev;
31
32 if (ep->busyPrev)
33 ep->busyPrev->busyNext = ep->busyNext;
34 else
35 memPool.tdQueueStart[ep->inTdQueue - 1] = ep->busyNext;
36
37 ep->inTdQueue = 0;
38}

40void enqueueEndpoint(Endpoint *ep, u32 listType)
41{
42 if (ep->inTdQueue)
43 return;
44
45 ep->busyNext = NULL;
46 ep->busyPrev = memPool.tdQueueEnd[listType - 1];
47 if (memPool.tdQueueEnd[listType - 1])
48 memPool.tdQueueEnd[listType - 1]->busyNext = ep;
49 else
50 memPool.tdQueueStart[listType - 1] = ep;
51 memPool.tdQueueEnd[listType - 1] = ep;
52
53 ep->inTdQueue = listType;
54}

67int setupControlTransfer(Endpoint *ep)
68{
69 HcTD *statusTd, *tailTd, *dataTd = NULL;
70 IoRequest *curIoReq = ep->ioReqListStart;
71
72 if (ep->hcEd.tdTail && !ED_HALTED(ep->hcEd) && !ED_SKIPPED(ep->hcEd) && curIoReq) {
73 if (curIoReq->destPtr && curIoReq->length) {
74 dataTd = allocTd();
75 if (!dataTd) {
76 enqueueEndpoint(ep, GENTD_QUEUE);
77 return 0;
78 }
79 }
80 statusTd = allocTd();
81 tailTd = allocTd();
82
83 if (!statusTd || !tailTd) {
84 freeTd(statusTd);
85 freeTd(tailTd);
86 freeTd(dataTd);
87
88 enqueueEndpoint(ep, GENTD_QUEUE);
89 return 0;
90 }
91
92 if (curIoReq->next)
93 curIoReq->next->prev = curIoReq->prev;
94 else
95 ep->ioReqListEnd = curIoReq->prev;
96
97 if (curIoReq->prev)
98 curIoReq->prev->next = curIoReq->next;
99 else
100 ep->ioReqListStart = curIoReq->next;
101
102 // first stage: setup
103 ep->hcEd.tdTail->HcArea = TD_HCAREA(USB_RC_NOTACCESSED, 2, 7, TD_SETUP, 0) << 16;
104 ep->hcEd.tdTail->curBufPtr = &curIoReq->devReq;
105 ep->hcEd.tdTail->bufferEnd = ((u8 *)&curIoReq->devReq) + sizeof(UsbDeviceRequest) - 1;
106
107 memPool.hcTdToIoReqLUT[ep->hcEd.tdTail - memPool.hcTdBuf] = curIoReq;
108
109 // second stage: data
110 if (dataTd) {
111 ep->hcEd.tdTail->next = dataTd;
112
113 if (curIoReq->devReq.requesttype & USB_DIR_IN)
114 dataTd->HcArea = TD_HCAREA(USB_RC_NOTACCESSED, 3, 7, TD_IN, 1) << 16;
115 else
116 dataTd->HcArea = TD_HCAREA(USB_RC_NOTACCESSED, 3, 7, TD_OUT, 1) << 16;
117
118 dataTd->curBufPtr = curIoReq->destPtr;
119 dataTd->bufferEnd = (u8 *)curIoReq->destPtr + curIoReq->length - 1;
120 dataTd->next = statusTd;
121
122 memPool.hcTdToIoReqLUT[dataTd - memPool.hcTdBuf] = curIoReq;
123 } else
124 ep->hcEd.tdTail->next = statusTd;
125
126 // third stage: status
127 if (curIoReq->devReq.requesttype & USB_DIR_IN)
128 statusTd->HcArea = TD_HCAREA(USB_RC_NOTACCESSED, 3, 0, TD_OUT, 0) << 16;
129 else
130 statusTd->HcArea = TD_HCAREA(USB_RC_NOTACCESSED, 3, 0, TD_IN, 0) << 16;
131
132 statusTd->curBufPtr = NULL;
133 statusTd->bufferEnd = NULL;
134 statusTd->next = tailTd;
135 memPool.hcTdToIoReqLUT[statusTd - memPool.hcTdBuf] = curIoReq;
136
137 ep->hcEd.tdTail = tailTd;
138
139 memPool.ohciRegs->HcCommandStatus |= OHCI_COM_CLF; // control list filled
140
141 // remove endpoint from busy list if there are no IoRequests left
142 if (!ep->ioReqListStart)
143 removeEndpointFromQueue(ep);
144 return 1;
145 } else {
146 // endpoint error
147 removeEndpointFromQueue(ep);
148 return 0;
149 }
150}

152int setupIsocronTransfer(Endpoint *ep)
153{
154 IoRequest *curIoReq = ep->ioReqListStart;
155
156 HcED *ed = &ep->hcEd;
157 HcIsoTD *curTd = (HcIsoTD *)ed->tdTail;
158 HcIsoTD *newTd;
159
160 u32 frameNo;
161
162 if (ep->hcEd.tdTail && !ED_HALTED(ep->hcEd) && !ED_SKIPPED(ep->hcEd) && curIoReq) {
163 newTd = allocIsoTd();
164 if (!newTd) {
165 enqueueEndpoint(ep, ISOTD_QUEUE);
166 return 0;
167 }
168
169 if (curIoReq->next)
170 curIoReq->next->prev = curIoReq->prev;
171 else
172 ep->ioReqListEnd = curIoReq->prev;
173
174 if (curIoReq->prev)
175 curIoReq->prev->next = curIoReq->next;
176 else
177 ep->ioReqListStart = curIoReq->next;
178
179 if (ed->tdTail == (HcTD *)((u32)ed->tdHead & ~0xF))
180 frameNo = memPool.hcHCCA->FrameNumber + 2;
181 else
182 frameNo = ep->isochronLastFrameNum;
183
184 frameNo = (frameNo + curIoReq->waitFrames) & 0xFFFF;
185
186 ep->isochronLastFrameNum = frameNo + 1;
187
188 curTd->hcArea = (USB_RC_NOTACCESSED << 28) | frameNo;
189 curTd->bufferPage0 = (void *)((u32)curIoReq->destPtr & ~0xFFF);
190 curTd->next = newTd;
191
192 if (curIoReq->destPtr && curIoReq->length)
193 curTd->bufferEnd = (u8 *)curIoReq->destPtr + curIoReq->length - 1;
194 else
195 curTd->bufferEnd = NULL;
196
197 curTd->psw[0] = (USB_RC_NOTACCESSED << 12) | ((u32)curIoReq->destPtr & 0xFFF);
198
199 memPool.hcIsoTdToIoReqLUT[curTd - memPool.hcIsoTdBuf] = curIoReq;
200
201 ed->tdTail = (HcTD *)newTd;
202
203 // remove endpoint from busy list if there are no IoRequests left
204 if (!ep->ioReqListStart)
205 removeEndpointFromQueue(ep);
206 return 1;
207 } else {
208 // endpoint error
209 removeEndpointFromQueue(ep);
210 return 0;
211 }
212}

214int setupBulkTransfer(Endpoint *ep)
215{
216 IoRequest *curIoReq = ep->ioReqListStart;
217
218 HcED *ed = &ep->hcEd;
219 HcTD *curTd = ed->tdTail;
220 HcTD *newTd;
221
222 if (ep->hcEd.tdTail && !ED_HALTED(ep->hcEd) && !ED_SKIPPED(ep->hcEd) && curIoReq) {
223 newTd = allocTd();
224 if (!newTd) {
225 enqueueEndpoint(ep, GENTD_QUEUE);
226 return 0;
227 }
228
229 if (curIoReq->next)
230 curIoReq->next->prev = curIoReq->prev;
231 else
232 ep->ioReqListEnd = curIoReq->prev;
233
234 if (curIoReq->prev)
235 curIoReq->prev->next = curIoReq->next;
236 else
237 ep->ioReqListStart = curIoReq->next;
238
239 curTd->HcArea = TD_HCAREA(USB_RC_NOTACCESSED, 0, 0, 3, 1) << 16;
240 curTd->next = newTd;
241 curTd->curBufPtr = curIoReq->destPtr;
242
243 if (curIoReq->destPtr && curIoReq->length)
244 curTd->bufferEnd = (u8 *)curIoReq->destPtr + curIoReq->length - 1;
245 else
246 curTd->bufferEnd = NULL;
247
248 memPool.hcTdToIoReqLUT[curTd - memPool.hcTdBuf] = curIoReq;
249
250 ed->tdTail = newTd;
251
252 if (ep->endpointType == TYPE_BULK)
253 memPool.ohciRegs->HcCommandStatus |= OHCI_COM_BLF; // Bulk List Filled
254
255 // remove endpoint from busy list if there are no IoRequests left
256 if (!ep->ioReqListStart)
257 removeEndpointFromQueue(ep);
258 return 1;
259 } else {
260 // endpoint error
261 dbg_printf("ERROR Endpoint error\n");
262 removeEndpointFromQueue(ep);
263 return 0;
264 }

267void handleIoReqList(Endpoint *ep)
268{
269 if (ep->endpointType == TYPE_CONTROL)
270 setupControlTransfer(ep);
271 else if (ep->endpointType == TYPE_ISOCHRON)
272 setupIsocronTransfer(ep);
273 else // bulk or interrupt
274 setupBulkTransfer(ep);
275}

277int attachIoReqToEndpoint(Endpoint *ep, IoRequest *req, void *destdata, u16 length, void *callback)
278{
279 if (!ep->correspDevice)
280 return USB_RC_BUSY;
281 if (req->busyFlag)
282 return USB_RC_BUSY;
283
284 req->busyFlag = 1;
285 req->correspEndpoint = ep;
286 req->destPtr = destdata;
287 req->length = length;
288 req->resultCode = 0;
289 req->transferedBytes = 0;
290 req->callbackProc = callback;
291
292 req->next = NULL;
293 req->prev = ep->ioReqListEnd;
294 if (ep->ioReqListEnd)
295 ep->ioReqListEnd->next = req;
296 else
297 ep->ioReqListStart = req;
298 ep->ioReqListEnd = req;
299
300 handleIoReqList(ep);
301 return 0;
302}
