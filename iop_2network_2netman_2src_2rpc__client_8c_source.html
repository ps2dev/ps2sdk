145  if (NumFramesInQueue > 0)

150  do{

154  if (mode == 0)

161  }while(res == 0);

163  NumFramesInQueue = 0;
164  }

178 static void EnQFrame(const void *frame, unsigned int length)
179 {
180  SifDmaTransfer_t *dmat;
181  struct NetManBD *bd;
182  iop_sys_clock_t clock;
183 
184  //Cancel any ongoing callbacks.
185  CancelAlarm(&FrameSendCB, NULL);
186 
187  if (NumFramesInQueue >= NETMAN_FRAME_GROUP_SIZE)
188  { /* If there are already sufficient frames, the frames can be sent right away.
189  This may happen here if sending failed within the interrupt callback and there are more frames to send. */
190  sendFramesToEE(0);
191  }
192 
193  //No need to wait for a free spot to appear, as Alloc already took care of that.
194  bd = &FrameBufferStatus[EEFrameBufferWrPtr];
195 
196  //Record the frame length.
197  bd->length = length;
198 
199  //Prepare DMA transfer.
200  dmat = &dmatReqs[NumFramesInQueue * 2];
201  dmat[0].src = (void*)frame;
202  dmat[0].dest = bd->payload;
203  dmat[0].size = (length + 3) & ~3;
204  dmat[0].attr = 0;
205 
206  dmat[1].src = bd;
207  dmat[1].dest = &EEFrameBufferStatus[EEFrameBufferWrPtr];
208  dmat[1].size = sizeof(struct NetManBD);
209  dmat[1].attr = 0;
210  NumFramesInQueue++;
211 
212  //Increase the write (IOP -> EE) pointer by one place.
213  EEFrameBufferWrPtr = (EEFrameBufferWrPtr + 1) % NETMAN_RPC_BLOCK_SIZE;
214 
215  if (NumFramesInQueue >= NETMAN_FRAME_GROUP_SIZE)
216  { //If there are sufficient frames, the frames can be sent right away.
217  sendFramesToEE(0);
218  } else {
219  //Wait a while in case further frames can be grouped, to allow sceSifSetDma() to chain the requests together.
220  clock.lo = FRAME_GROUPING_INTERVAL;
221  clock.hi = 0;
222  SetAlarm(&clock, &FrameSendCB, NULL);
223  }
224 }
