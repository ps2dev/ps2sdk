147 {
148  iop_thread_t thread_param;
149  register int thread_id;
150  iop_library_t *libptr;

154 #ifdef SIO_DEBUG

156 #endif
157 #ifdef DEBUG
158  DPRINTF("mcserv: _start...\n");
159 #endif
160 
161  // Get mcman lib ptr
162  mcman_loaded = 0;
163  libptr = GetLoadcoreInternalData()->let_next;
164  while (libptr != 0) {
165  for (i=0; i<8; i++) {
166  if (libptr->name[i] != mcman_modname[i])
167  break;
168  }

170  mcman_loaded = 1;
171  break;
172  }

174  }

278 void *cb_rpc_S_0400(u32 fno, void *buf, int size)
279 {
280  // Rpc Callback function
281  int (*rpc_func)(void);
282  register int i;
283 
284  // Determine the RPC client based on the init command.
285  if (fno == MCSERV_RpcCmd[0][0][0]) // MCSERV
286  {
287  mcserv_interface_type = 0;
288  }
289  else if (fno == MCSERV_RpcCmd[1][0][0]) // XMCSERV
290  {
291  mcserv_interface_type = 1;
292  }
293 
294  for (i=0; i<sizeof(MCSERV_RpcCmd[0][0]) / sizeof(MCSERV_RpcCmd[0][0][0]); i++) { // retrieve correct function number for mcserv
295  if (fno == MCSERV_RpcCmd[mcserv_interface_type][0][i]) {
296  fno = MCSERV_RpcCmd[mcserv_interface_type][1][i];
297  break;
298  }
299  }
300  // If function is not found, return
301  if (fno >= sizeof(rpc_funcs_array) / sizeof(rpc_funcs_array[0]))
302  {
303  return (void *)&rpc_stat;
304  }
305 
306  // Get function pointer
307  rpc_func = (void *)rpc_funcs_array[fno];
308 
309  // Call needed rpc func
310  rpc_stat.result = rpc_func();
311 
312  pMcReplaceBadBlock();
313 
314  return (void *)&rpc_stat;
315 }

517 int _McRead(void *rpc_buf)
518 {
519  mcDescParam_t *dP = (mcDescParam_t *)rpc_buf;
520  register int file_offset, status;
521  register int left_to_read, size_readed, size_to_read;
522  mcEndParam_t eP;
523  SifDmaTransfer_t dmaStruct;
524  int intStatus;
525  void *eedata;
526 
527 #ifdef DEBUG
528  DPRINTF("mcserv: _McRead fd %d ee buffer addr %x size %d\n", dP->fd, (int)dP->buffer, dP->size);
529 #endif
530 
531  eP.size1 = 0;
532  eP.size2 = 0;
533  eP.dest1 = NULL;
534  eP.dest2 = NULL;
535 
536  size_readed = 0;
537  file_offset = 0;
538 
539  eedata = dP->buffer;
540 
541  eP.size1 = dP->size;
542 
543  if (dP->size > 16)
544  eP.size1 = (((int)(eedata)-1) & 0xfffffff0) - ((int)(eedata) - 16);
545 
546  eP.size2 = (dP->size - eP.size1) & 0x0f;
547  left_to_read = (dP->size - eP.size1) - eP.size2;
548 
549  if (eP.size2 != 0)
550  eP.dest2 = (void *)(eedata + eP.size1 + left_to_read);
551 
552  if (eP.size1 != 0) {
553  size_readed = pMcRead(dP->fd, eP.src1, eP.size1);
554 
555  if (size_readed < 0) {
556  eP.size1 = 0;
557  eP.size2 = 0;
558  goto dma_transfer2;
559  }
560  else {
561  file_offset = size_readed;
562  eP.dest1 = eedata;
563  eedata += size_readed;
564 
565  if (size_readed != eP.size1) {
566  eP.size1 = size_readed;
567  eP.size2 = 0;
568  size_readed = 0;
569  goto dma_transfer2;
570  }
571  else
572  size_readed = 0;
573  }
574  }
575 
576  while (left_to_read > 0) {
577 
578  size_to_read = left_to_read;
579 
580  if (left_to_read > MCSERV_BUFSIZE)
581  size_to_read = MCSERV_BUFSIZE;
582 
583  size_readed = pMcRead(dP->fd, mcserv_buf, size_to_read);
584 
585  if (size_readed < 0) {
586  eP.size2 = 0;
587  goto dma_transfer2;
588  }
589 
590  dmaStruct.src = (void *)mcserv_buf;
591  dmaStruct.dest = (void *)eedata;
592  dmaStruct.size = size_readed;
593  dmaStruct.attr = 0;
594 
595  CpuSuspendIntr(&intStatus);
596  sceSifSetDma(&dmaStruct, 1);
597 
598  file_offset += size_readed;
599  left_to_read -= size_readed;
600 
601  CpuResumeIntr(intStatus);
602 
603  eedata += size_readed;
604 
605  if (size_to_read != size_readed) {
606  eP.size2 = 0;
607  size_readed = 0;
608  goto dma_transfer2;
609  }
610  size_readed = 0;
611  }
612 
613  if (eP.size2 == 0)
614  goto dma_transfer2;
615 
616  size_readed = pMcRead(dP->fd, eP.src2, eP.size2);
617 
618  if (size_readed < 0) {
619  eP.size2 = 0;
620  goto dma_transfer2;
621  }
622 
623  file_offset += size_readed;
624  eP.size2 = size_readed;
625  size_readed = 0;
626 
627 dma_transfer2:
628  dmaStruct.src = (void *)&eP;
629  dmaStruct.dest = (void *)dP->param;
630  dmaStruct.size = sizeof (mcEndParam_t);
631  dmaStruct.attr = 0;
632 
633  CpuSuspendIntr(&intStatus);
634  status = sceSifSetDma(&dmaStruct, 1);
635  CpuResumeIntr(intStatus);
636 
637  while (sceSifDmaStat(status) >= 0)
638  DelayThread(100);
639 
640  if (size_readed != 0)
641  return size_readed;
642 
643  return file_offset;
644 }

719
