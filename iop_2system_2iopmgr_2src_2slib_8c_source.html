37iop_library_t *slib_get_lib_by_name(const char *name)
38{
39 iop_library_t *libptr;
40 int len = strlen(name) + 1;
41
42 libptr = GetLoadcoreInternalData()->let_next;
43 while ((libptr != 0))
44 {
45 if (!memcmp(libptr->name, name, len))
46 return libptr;
47 libptr = libptr->prev;
48 }
49 return 0;
50}
